通信报文

  • 上线通知

{
    cmd: "online",
    name: "account_name"
}

  • 上线回应

{
    cmd: "onlineack",
    name: "account_name"
}

  • 修改账户名

{
    cmd: "changename",
    name: "account_name"
}

  • 聊天

{
    cmd: "chat",
    broadcast: true,
    content: "msg-body",
    name: "account_name"  
}

  • 发送文件

文件请求
{
    cmd: sendfile
    name: xxx.avi
    id: sessionid
}
 
文件请求响应
{
    cmd: sendfileack
    name: xxx.avi
    result: ok
    id: sessionid
}
{
    cmd: sendfileack
    name: xxx.avi
    result: cancel
    id: sessionid
